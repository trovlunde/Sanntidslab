163 static inline uint32_t __sd_nvic_app_accessible_irq(IRQn_Type IRQn)
164 {
165  if (IRQn < 32)
166  {
167  return ((1UL<<IRQn) & __NRF_NVIC_APP_IRQS_0) != 0;
168  }
169 #ifdef NRF52
170  else if (IRQn < 64)
171  {
172  return ((1UL<<(IRQn-32)) & __NRF_NVIC_APP_IRQS_1) != 0;
173  }
174 #endif
175  else
176  {
177  return 1;
178  }
179 }

203  || priority == 4
204  || priority == 5
205  )
206  {
207  return 0;
208  }
209 #endif
210  return 1;
211 }

229 static inline uint32_t sd_nvic_EnableIRQ(IRQn_Type IRQn)
230 {
231  if (!__sd_nvic_app_accessible_irq(IRQn))
232  {
233  return NRF_ERROR_SOC_NVIC_INTERRUPT_NOT_AVAILABLE;
234  }
235  if (!__sd_nvic_is_app_accessible_priority(NVIC_GetPriority(IRQn)))
236  {
237  return NRF_ERROR_SOC_NVIC_INTERRUPT_PRIORITY_NOT_ALLOWED;
238  }
239 
240  if (nrf_nvic_state.__cr_flag)
241  {
242  nrf_nvic_state.__irq_masks[(uint32_t)((int32_t)IRQn) >> 5] |= (uint32_t)(1 << ((uint32_t)((int32_t)IRQn) & (uint32_t)0x1F));
243  }
244  else
245  {
246  NVIC_EnableIRQ(IRQn);
247  }
248  return NRF_SUCCESS;
249 }

425 static inline uint32_t sd_nvic_critical_region_enter(uint8_t * p_is_nested_critical_region)
426 {
427  int was_masked = __sd_nvic_irq_disable();
428  if (!nrf_nvic_state.__cr_flag)
429  {
430  nrf_nvic_state.__cr_flag = 1;
431  nrf_nvic_state.__irq_masks[0] = ( NVIC->ICER[0] & __NRF_NVIC_APP_IRQS_0 );
432  NVIC->ICER[0] = __NRF_NVIC_APP_IRQS_0;
433  #ifdef NRF52
434  nrf_nvic_state.__irq_masks[1] = ( NVIC->ICER[1] & __NRF_NVIC_APP_IRQS_1 );
435  NVIC->ICER[1] = __NRF_NVIC_APP_IRQS_1;
436  #endif
437  *p_is_nested_critical_region = 0;
438  }
439  else
440  {
441  *p_is_nested_critical_region = 1;
442  }
443  if (!was_masked)
444  {
445  __sd_nvic_irq_enable();
446  }
447  return NRF_SUCCESS;
448 }
